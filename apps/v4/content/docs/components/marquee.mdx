## Usage

```tsx
import { Marquee } from "@/components/ui/marquee"

function LogoMarquee() {
  return (
    <Marquee pauseOnHover>
      <Logo1 />
      <Logo2 />
      <Logo3 />
    </Marquee>
  )
}
```

## Props

| Prop           | Type      | Default | Description                               |
| -------------- | --------- | ------- | ----------------------------------------- |
| `reverse`      | `boolean` | `false` | Reverse the scroll direction              |
| `pauseOnHover` | `boolean` | `false` | Pause animation on hover                  |
| `vertical`     | `boolean` | `false` | Scroll vertically instead of horizontally |
| `repeat`       | `number`  | `4`     | Number of times to repeat the content     |
| `className`    | `string`  | -       | Additional CSS classes                    |

## Examples

### Horizontal Marquee

```tsx
<Marquee>
  {logos.map((logo) => (
    <img key={logo.name} src={logo.src} alt={logo.name} className="h-8" />
  ))}
</Marquee>
```

### Vertical Marquee

```tsx
<Marquee vertical className="h-[300px]">
  {testimonials.map((t) => (
    <TestimonialCard key={t.id} {...t} />
  ))}
</Marquee>
```

### Reversed Direction

```tsx
<Marquee reverse>
  {items.map((item) => (
    <Item key={item.id} {...item} />
  ))}
</Marquee>
```

### Custom Speed

Control the speed using CSS variables:

```tsx
<Marquee className="[--duration:20s]">
  {items.map((item) => (
    <Item key={item.id} {...item} />
  ))}
</Marquee>
```

### Custom Gap

```tsx
<Marquee className="[--gap:2rem]">
  {items.map((item) => (
    <Item key={item.id} {...item} />
  ))}
</Marquee>
```
