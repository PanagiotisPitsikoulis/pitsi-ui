The Texture Button component provides buttons with multi-layered gradient textures that create a tactile, 3D-like appearance. Works great for call-to-action buttons and premium interfaces.

## Usage

```tsx
import { TextureButton } from "@/components/ui/texture-button"

function Example() {
  return <TextureButton>Click me</TextureButton>
}
```

## Props

| Prop      | Type                                                                           | Default     | Description             |
| --------- | ------------------------------------------------------------------------------ | ----------- | ----------------------- |
| `variant` | `"primary" \| "secondary" \| "accent" \| "destructive" \| "minimal" \| "icon"` | `"primary"` | Button variant          |
| `size`    | `"default" \| "sm" \| "lg" \| "icon"`                                          | `"default"` | Button size             |
| `asChild` | `boolean`                                                                      | `false`     | Render as child element |

## Examples

### Primary Button

```tsx
<TextureButton variant="primary">Get Started</TextureButton>
```

### Secondary Button

```tsx
<TextureButton variant="secondary">Learn More</TextureButton>
```

### Accent Button

```tsx
<TextureButton variant="accent">Upgrade Now</TextureButton>
```

### Destructive Button

```tsx
<TextureButton variant="destructive">Delete</TextureButton>
```

### Minimal Button

```tsx
<TextureButton variant="minimal">Cancel</TextureButton>
```

### Icon Button

```tsx
import { Plus } from "lucide-react"

;<TextureButton variant="icon" size="icon">
  <Plus className="h-4 w-4" />
</TextureButton>
```

### Small Size

```tsx
<TextureButton size="sm">Small Button</TextureButton>
```

### Large Size

```tsx
<TextureButton size="lg">Large Button</TextureButton>
```

### As Link

```tsx
import Link from "next/link"

;<TextureButton asChild>
  <Link href="/dashboard">Go to Dashboard</Link>
</TextureButton>
```
